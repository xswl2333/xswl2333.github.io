//函数调用运算符重载
class person
{
public:
    void operator()(string test)
    {
        cout << test << endl;
    }
};
class addnum
{
public:
    int operator()(int a,int b)//仿函数创建灵活
    {
        return a + b;
    }
};
void test6()
{
    person hum;
    hum("tom");//调用像函数，称为仿函数
}

void test7()
{
    addnum p;//创建对象
    cout << p(1, 2) << endl;

}
int main()
{
    test7();
    return 0;
}

构造函数和析构函数
