struct Personbook//通讯录结构体变量

void addPerson(Personbook *abs)
{
if (abs->m_size==MAX)//目录逐条输入，等于MAX时立即打断

}

//检测联系人是否存在？返回联系人数组中位置：不存在返回-1
int exitPerson(Personbook *abs,string name)
{
for (int i = 0; i < abs->m_size; i++)
{
if (name == abs->array[i].p_name)
{
return i;
}
}
return -1;//放到for循环外部，等到全部遍历完再返回，否则数组为空，开始会返回0，
}

{
